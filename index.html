function parseOzon(rr){ if(!rr || !rr.length) return []; const H = rr._headers; const c_date = col(H,'Дата начисления'); const c_grp = col(H,'Группа услуг'); const c_typ = col(H,'Тип начисления'); const c_art = col(H,'Артикул'); const c_qty = col(H,'Количество'); const c_sum = col(H,'Сумма итого, руб.','Сумма итого руб','Сумма итого'); if(!c_sum || !c_date) return []; // ===== Переменные (экономика) ===== const eH = (raw.ekon && raw.ekon._headers) || []; const e_code = col(eH,'Код латиница','Код латиница'); const e_rs = col(eH,'Расход-ШТ','Расход-Шт','Расход ШТ'); const ekonByCode = {}; if(e_code && e_rs){ (raw.ekon || []).forEach(r=>{ const k = lnk(r[e_code]); if(k) ekonByCode[k] = num(r[e_rs]); }); } function cleanCode(art){ let a = String(art || '').replace('_XL','_1XL'); let b = a.split('_')[0]; let c = b.split('0')[0]; return c.replace( /TermSocks\.Al-ESS\.(black|blue|grey|pink|purp)\./i, 'TermSocks.Al-ESS.' ); } const map = {}; rr.forEach(r=>{ const d = pdate(r[c_date]); if(!d) return; const grp = String(r[c_grp] || '').trim(); const typ = String(r[c_typ] || '').trim(); const art = String(r[c_art] || '').trim(); const qty = Math.abs(num(r[c_qty])); const s = num(r[c_sum]); const base = art.split('_')[0] || art; const rec = artByPostL.get(lnk(base)) || artByPostA.get(nk(base)); const paG = rec ? rec.paG : base; const key = d.getFullYear() + '-' + (d.getMonth()+1) + '|' + paG; const o = map[key] || (map[key] = { y:d.getFullYear(), m:d.getMonth()+1, paG, code:cleanCode(art), salesQty:0, orderQty:0, salesRub:0, bonusRub:0, returnRub:0, partnerRub:0, log:0, rek:0, voz:0 }); // ---------- ШТУКИ ---------- if(typ === 'Выручка') o.salesQty += qty; if(typ === 'Логистика') o.orderQty += qty; // ---------- РУБЛИ ---------- if(typ === 'Выручка') o.salesRub += s; if(typ === 'Баллы за скидки') o.bonusRub += s; if(typ === 'Возврат выручки') o.returnRub += s; if(typ === 'Программы партнёров') o.partnerRub += s; // ---------- Остальные расходы ---------- if(grp === 'Вознаграждение Ozon') o.kom = (o.kom || 0) + s; else if( grp === 'Услуги доставки' || grp === 'Услуги партнёров' || grp === 'Услуги FBO' || grp === 'Другие услуги и штрафы' ) o.log += s; else if(grp === 'Продвижение и реклама') o.rek += s; else if( grp === 'Возвраты' || grp === 'Компенсации и декомпенсации' ) o.voz += s; }); return Object.values(map).map(o=>{ // Выкуп, руб (как в Power BI) const vykr = Math.round( o.salesRub + o.bonusRub + o.returnRub + o.partnerRub ); // Цена нам const price = o.salesQty ? vykr / o.salesQty : 0; // Заказ, руб const zakr = Math.round( price * o.orderQty ); // Логистика const dost = Math.round( -(o.log + o.voz) ); // Реклама const rek = Math.round( -o.rek ); // Переменные const perem = Math.round( (ekonByCode[lnk(o.code)] || 0) * o.salesQty ); // Налог const nalog = Math.round( vykr * 0.07 ); return { paG : o.paG, y : o.y, m : o.m, // ======== как в Power BI ======== zaks : o.orderQty, vyks : o.salesQty, zakr : zakr, vykr : vykr, // ================================ rek : rek, post : 0, nalog : nalog, hran : 0, dost : dost, perem : perem }; }); }
